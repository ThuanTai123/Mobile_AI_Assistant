import React, { useState, useRef, useEffect } from 'react';
import {
  View,
  Text,
  TextInput,
  FlatList,
  TouchableOpacity,
  StyleSheet,
  KeyboardAvoidingView,
  Platform,
  Vibration,
  SafeAreaView,
} from 'react-native';
import {
  toggleFlashlight,
  toggleNotification,
  openNavigationBar,
  increaseVolume,
  decreaseVolume,
} from './DeviceControls';
import { Ionicons } from '@expo/vector-icons';
import { checkAndOpenApp } from './AppLauncher'; // Ä‘Æ°á»ng dáº«n theo cáº¥u trÃºc dá»± Ã¡n cá»§a báº¡n
import * as Speech from 'expo-speech';
import { sendMessageToBot } from '../api/chat';
import * as Notifications from 'expo-notifications';

interface Message {
  id: number;
  text: string;
  sender: 'user' | 'bot';
}

const generateId = () => Date.now() + Math.floor(Math.random() * 1000);

const ChatScreen = () => {
  const [messages, setMessages] = useState<Message[]>([]);
  const [inputText, setInputText] = useState('');
  const flatListRef = useRef<FlatList>(null);

  useEffect(() => {
    requestPermissions();

    Notifications.setNotificationHandler({
      handleNotification: async () => ({
        shouldShowAlert: false,
        shouldPlaySound: true,  
        shouldSetBadge: false,
        shouldShowBanner: true,
        shouldShowList: true,
      }),
    });
  }, []);

  const requestPermissions = async () => {
    const { status } = await Notifications.requestPermissionsAsync();
    if (status !== 'granted') {
      alert('Báº¡n cáº§n cáº¥p quyá»n thÃ´ng bÃ¡o Ä‘á»ƒ nháº­n nháº¯c nhá»Ÿ');
    }
  };

  const scrollToBottom = () => {
    flatListRef.current?.scrollToEnd({ animated: true });
  };

  const handleDeviceCommand = async (text: string): Promise<boolean> => {
    const lower = text.toLowerCase();

    if (lower.includes('má»Ÿ thanh Ä‘iá»u hÆ°á»›ng') || lower.includes('thoÃ¡t app')) {
      openNavigationBar();
      return true;
    }
    if (lower.includes('báº­t Ä‘Ã¨n flash')) {
      await toggleFlashlight(true);
      return true;
    }
    if (lower.includes('táº¯t Ä‘Ã¨n flash')) {
      await toggleFlashlight(false);
      return true;
    }
    if (lower.includes('báº­t thÃ´ng bÃ¡o')) {
      toggleNotification(true);
      return true;
    }
    if (lower.includes('táº¯t thÃ´ng bÃ¡o')) {
      toggleNotification(false);
      return true;
    }
    if (lower.includes('tÄƒng Ã¢m lÆ°á»£ng')) {
      increaseVolume();
      return true;
    }
    if (lower.includes('giáº£m Ã¢m lÆ°á»£ng')) {
      decreaseVolume();
      return true;
    }

    return false;
  };


  const handleSend = async (overrideText?: string) => {
    const textToSend = overrideText || inputText.trim();
    if (!textToSend) return;
    
    const isDeviceCommand = await handleDeviceCommand(textToSend);
    if (isDeviceCommand) {
      setMessages((prev) => [
        ...prev,
        {
          id: generateId(),
          text: `ðŸ¤– ÄÃ£ thá»±c hiá»‡n lá»‡nh: "${textToSend}"`,
          sender: 'bot',
        },
      ]);
      setInputText('');
      scrollToBottom();
      return;
    }
    
    const { opened, appName } = await checkAndOpenApp(textToSend);
      if (opened) {
        setMessages((prev) => [
          ...prev,
          {
            id: generateId(),
            text: `ÄÃ£ má»Ÿ á»©ng dá»¥ng ${appName} cho báº¡n.`,
            sender: 'bot',
          },
        ]
      );
      return;
    }

    const userMessage: Message = {
      id: generateId(),      text: textToSend,
      sender: 'user',
    };
    

    setMessages((prev) => [...prev, userMessage]);
    setInputText('');
    scrollToBottom();

    const botResponse = await sendMessageToBot(textToSend);

     // Kiá»ƒm tra náº¿u lÃ  lá»‡nh nháº¯c, thÃ¬ trÃ­ch sá»‘ giÃ¢y vÃ  lÃªn lá»‹ch
    const isReminder = /Ä‘Ã£ táº¡o nháº¯c/i.test(botResponse.reply);
    if (isReminder) {
      const match = textToSend.match(/(\d+)\s*(giÃ¢y|giay|seconds?)/i);
      if (match) {
        const delaySeconds = parseInt(match[1]);
        if (!isNaN(delaySeconds)) {
          await scheduleReminderNotification(delaySeconds, 'â° ÄÃ£ táº¡o nháº¯c nhá»Ÿ ' + textToSend);
        }
      }
    }
    
    const botMessage: Message = {
      id: generateId(),      text: botResponse.reply,
      sender: 'bot',
    };

    setMessages((prev) => [...prev, botMessage]);
    scrollToBottom();

    // Äá»c pháº£n há»“i báº±ng giá»ng nÃ³i
    if (!isReminder){
      Speech.speak(botResponse.reply, {
        language: 'vi-VN',
        pitch: 1,
        rate: 1,
      });
    }
    // Náº¿u pháº£n há»“i cÃ³ dáº¡ng nháº¯c viá»‡c, tÃ¬m sá»‘ giÃ¢y Ä‘á»ƒ Ä‘áº·t thÃ´ng bÃ¡o
    if (botResponse.reply.toLowerCase().includes('Ä‘Ã£ táº¡o nháº¯c')) {
      const match = textToSend.match(/(\d+)\s*(giÃ¢y|giay|seconds?)/i);
      if (match) {
        const delaySeconds = parseInt(match[1]);
        if (!isNaN(delaySeconds)) {
          scheduleReminderNotification(delaySeconds, 'â° Nháº¯c báº¡n: ' + textToSend);
        }
      }
    }
  };

  const scheduleReminderNotification = async (delaySeconds: number, message: string) => {
    await Notifications.scheduleNotificationAsync({
      content: {
        title: 'Nháº¯c nhá»Ÿ',
        body: message,
        sound: true,
      },
      trigger: {
        seconds: delaySeconds,
        repeats: false,
      } as Notifications.TimeIntervalTriggerInput
    });
  };


  return (
    <SafeAreaView style={{ flex: 1 }}>
      <KeyboardAvoidingView
        style={styles.container}
        behavior={Platform.OS === 'ios' ? 'padding' : undefined}
        keyboardVerticalOffset={Platform.OS === 'ios' ? 90 : 0}
      >
        <FlatList
          ref={flatListRef}
          data={messages}
          keyExtractor={(item) => item.id.toString()}
          renderItem={({ item }) => (
            <View style={item.sender === 'user' ? styles.userMsg : styles.botMsg}>
              <Text style={styles.msgText}>{item.text}</Text>
            </View>
          )}
          contentContainerStyle={styles.messageList}
          onContentSizeChange={scrollToBottom}
          keyboardShouldPersistTaps="handled"
        />



        <View style={[styles.inputContainer, { paddingBottom: Platform.OS === 'android' ? 40 : 0 }]}>
          <TextInput
            style={styles.textInput}
            value={inputText}
            onChangeText={setInputText}
            placeholder="Nháº­p tin nháº¯n..."
            returnKeyType="send"
            onSubmitEditing={() => handleSend()}
          />
          <TouchableOpacity onPress={() => handleSend()} style={styles.iconButton}>
            <Ionicons name="send" size={28} color="blue" />
          </TouchableOpacity>

          <TouchableOpacity style={styles.iconButton}>
            <Ionicons name="mic-outline" size={36} color="gray" />
          </TouchableOpacity>
        </View>
      </KeyboardAvoidingView>
    </SafeAreaView>
  );
};

export default ChatScreen;

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#fff',
  },
  messageList: {
    padding: 10,
    paddingBottom: 80,
  },
  userMsg: {
    alignSelf: 'flex-end',
    backgroundColor: '#dcf8c6',
    borderRadius: 12,

  },
  botMsg: {

    borderRadius: 12,
    padding: 10,

  },
  msgText: {

  },
  inputContainer: {

  },
  textInput: {

    marginRight: 8,
  },
  iconButton: {

  },
});
